## Queries

### 1. Display all employees with their department name
```sql
SELECT E.ENAME, D.DNAME
FROM EMPLOYEE E, DEPARTMENT D
WHERE E.DEPTNO = D.DEPTNO;
```

### 2. Employees whose manager name is JONES
```sql
SELECT E.ENAME, M.ENAME AS MANAGER
FROM EMPLOYEE E, EMPLOYEE M
WHERE E.MGR = M.EMPNO AND M.ENAME = 'JONES';
```

### 3. Employee details with dept, manager and grade
```sql
SELECT E.ENAME, E.JOB, D.DNAME, M.ENAME AS MANAGER, S.GRADE
FROM EMPLOYEE E, DEPARTMENT D, EMPLOYEE M, SALGRADE S
WHERE E.DEPTNO = D.DEPTNO
AND E.MGR = M.EMPNO(+)
AND E.SAL BETWEEN S.LOSAL AND S.HISAL
ORDER BY D.DNAME;
```

### 4. Employees except clerk sorted by salary
```sql
SELECT E.ENAME, E.JOB, E.SAL, S.GRADE, D.DNAME
FROM EMPLOYEE E, DEPARTMENT D, SALGRADE S
WHERE E.DEPTNO = D.DEPTNO
AND E.SAL BETWEEN S.LOSAL AND S.HISAL
AND E.JOB <> 'CLERK'
ORDER BY E.SAL DESC;
```

### 5. Employee with manager (including no manager)
```sql
SELECT E.ENAME, E.JOB, NVL(M.ENAME,'NO MANAGER') AS MANAGER
FROM EMPLOYEE E, EMPLOYEE M
WHERE E.MGR = M.EMPNO(+);
```

### 6. Employees earning 36000 annually or not clerks
```sql
SELECT E.ENAME, E.JOB, E.SAL*12 AS ANNUAL, E.DEPTNO, D.DNAME, S.GRADE
FROM EMPLOYEE E, DEPARTMENT D, SALGRADE S
WHERE E.DEPTNO = D.DEPTNO
AND E.SAL BETWEEN S.LOSAL AND S.HISAL
AND (E.SAL*12 = 36000 OR E.JOB <> 'CLERK');
```

### 7. Employees earning 30000 annually and not clerks
```sql
SELECT E.ENAME, E.JOB, E.SAL*12 AS ANNUAL, E.DEPTNO, D.DNAME, S.GRADE
FROM EMPLOYEE E, DEPARTMENT D, SALGRADE S
WHERE E.DEPTNO = D.DEPTNO
AND E.SAL BETWEEN S.LOSAL AND S.HISAL
AND E.SAL*12 = 30000
AND E.JOB <> 'CLERK';
```

### 8. Employees with manager details
```sql
SELECT E.ENAME, E.EMPNO,
NVL(M.ENAME,'NO MANAGER') AS MANAGER,
M.EMPNO AS MGR_NO
FROM EMPLOYEE E, EMPLOYEE M
WHERE E.MGR = M.EMPNO(+);
```

### 9. Dept name, dept no and sum of salary
```sql
SELECT D.DNAME, D.DEPTNO, SUM(E.SAL)
FROM EMPLOYEE E, DEPARTMENT D
WHERE E.DEPTNO = D.DEPTNO
GROUP BY D.DNAME, D.DEPTNO;
```

### 10. Employee number, name and department location
```sql
SELECT E.EMPNO, E.ENAME, D.LOC
FROM EMPLOYEE E, DEPARTMENT D
WHERE E.DEPTNO = D.DEPTNO;
```

### 11. Employee name and department name
```sql
SELECT E.ENAME, D.DNAME
FROM EMPLOYEE E, DEPARTMENT D
WHERE E.DEPTNO = D.DEPTNO;
```
